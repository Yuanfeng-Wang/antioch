71 template <typename Scalar>
72 Scalar F(const Scalar & x)
73 {
74  return 1.L + Antioch::ant_pow(pi,(float)1.5)/2.L * Antioch::ant_sqrt(x)
75  + (pi*pi/4.L + 2.L) * x
76  + Antioch::ant_pow(pi * x,(float)1.5);
77 }
