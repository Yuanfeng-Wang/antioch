41  /* Stockmayer potential from
42  Monchich & Mason 1961, surface[T][delta]
43 
44  In case of temperatures that are too high
45  (\f$T > 100 \eps_{\mathrm{LJ}}\f$), we extrapolate using the
46  following expression:
47  \f[
48  \left<\Omega^{(l,s)*}\right> \proto \sqrt[6]{T^{*}}
49  \f]
50  The proportionality factor is found using the last data point at \f$T^* = 100\f$
51 
52  Two points are added when extrapolating to \f$T^* + \Delta T^*\f$ is required,
53  \f$100 + \Delta T^*\f$ and \f$100 + frac{\Delta T^*}{2}\f$.
54  */

148  inline
149  void StockmayerPotential<CoeffType>::extrapolate_to(const StateType & T)
150  {
151  // !\todo better vectorization management
152  //
153  // we need to have something as a raw_max,
154  // probably in the form of rawify<StateType,Operation>()
155  // to avoid repetition. Note that two non equivalent
156  // strategies are possible here:
157  // - VectorType<raw_type<StateType> >: all raw_values are equivalent, looking for global Operation
158  // - Operation<StateType>() at each level, looking for local Operation at every level
159  CoeffType Delta_T = (max(T) - _max_reduced_T);
160 
161  // extrapolation required
162  if(Delta_T > 0)
163  {
164 
165  //resize
166  _omega_1_1.push_back(std::vector<CoeffType>(_delta_size)); // + Delta T/2
167  _omega_1_1.push_back(std::vector<CoeffType>(_delta_size)); // + Delta T
168  _omega_2_2.push_back(std::vector<CoeffType>(_delta_size)); // + Delta T/2
169  _omega_2_2.push_back(std::vector<CoeffType>(_delta_size)); // + Delta T
170 
171  _T_size += 2;
172  _T.push_back(_max_reduced_T + Delta_T/2.);
173  _T.push_back(_max_reduced_T + Delta_T);
174  _logT.push_back(ant_log(_T[_T_size - 2]));
175  _logT.push_back(ant_log(_T[_T_size - 1]));
176 
177  for(unsigned int d = 0; d < _delta_size; d++)
178  {
179  CoeffType K = _omega_1_1[_T_size - 3][d] / (ant_pow(_T[_T_size - 3],CoeffType(1.L/6.L)));
180  _omega_1_1[_T_size - 2][d] = K * _T[_T_size - 2];
181  _omega_1_1[_T_size - 1][d] = K * _T[_T_size - 1];
182 
183  K = _omega_2_2[_T_size - 3][d] / (ant_pow(_T[_T_size - 3],CoeffType(1.L/6.L)));
184  _omega_2_2[_T_size - 2][d] = K * _T[_T_size - 2];
185  _omega_2_2[_T_size - 1][d] = K * _T[_T_size - 1];
186  }
187  }
188  }
